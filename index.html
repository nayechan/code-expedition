Google Apps Script 코드 (복사 후 스크립트 에디터에 붙여넣기)
const MEMBER_COL = { '김호민': 4, '나예찬': 5, '이준용': 6, '장재범': 7, '정찬호': 8, '최진혁': 9 }; function doGet(e) { const p = e.parameter; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; const range = sheet.getDataRange(); const values = range.getValues(); const display = range.getDisplayValues(); const col = MEMBER_COL[p.member]; if (!col) return json({ ok: false, error: 'Unknown member' }); for (let i = 1; i < display.length; i++) { if (normDate(display[i][0]) === p.date) { sheet.getRange(i + 1, col).setValue(parseFloat(p.solveTime)); return json({ ok: true }); } } return json({ ok: false, error: 'Date not found: ' + p.date }); } function normDate(v) { const m = String(v).match(/(\d{4})\.\s*(\d{1,2})\.\s*(\d{1,2})/); return m ? `${m[1]}-${m[2].padStart(2,'0')}-${m[3].padStart(2,'0')}` : ''; } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
